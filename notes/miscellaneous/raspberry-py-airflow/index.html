Airflow

export AIRFLOW_HOME=~/airflow
# TMPDIR={dir} pip install --cache-dir={dir} --build {dir} {package-name}
TMPDIR=/var/tmp pip install --cache-dir=/var/tmp --build /var/tmp apache-airflow

cd ~/airflow
airflow init db
airflow check

airflow create_user [-h] [-r ROLE] [-u USERNAME] [-e EMAIL] [-f FIRSTNAME]
                    [-l LASTNAME] [-p PASSWORD] [--use_random_password]

# start airflow
airflow webserver & airflow scheduler
